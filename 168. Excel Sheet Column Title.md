public class Solution {
    public String convertToTitle(int n) {
        if (n <= 0) return "";
        StringBuilder title = new StringBuilder();
        while (n > 0) {
            n--;
            int r = n % 26;
            title.insert(0, (char)('A' + r));
            n = n / 26;
        }
        return title.toString();
    }
}
